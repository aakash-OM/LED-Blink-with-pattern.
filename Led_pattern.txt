void setup()
{
pinMode(12,OUTPUT);
pinMode(11,OUTPUT);
pinMode(9,OUTPUT);
}

void loop()
{
digitalWrite(12,HIGH);
digitalWrite(11,LOW);
digitalWrite(9,LOW);
delay(1000);

digitalWrite(12,LOW);
digitalWrite(11,HIGH);
digitalWrite(9,LOW);
delay(1000);
 
digitalWrite(12,LOW);
digitalWrite(11,LOW);
digitalWrite(9,HIGH);
delay(1000);  


}
